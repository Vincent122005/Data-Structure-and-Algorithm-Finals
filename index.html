Infix to Postfix Conversion

Convert mathematical expressions using stack-based algorithm

Conversion Algorithm

def infix_to_postfix(expression): precedence = {'+': 1, '-': 1, '*': 2, '/': 2, '^': 3} stack = [] output = [] for char in expression: if char.isalnum(): output.append(char) elif char == '(': stack.append(char) elif char == ')': while stack and stack[-1] != '(': output.append(stack.pop()) stack.pop() else: while (stack and stack[-1] != '(' and precedence.get(char, 0) <= precedence.get(stack[-1], 0)): output.append(stack.pop()) stack.append(char) while stack: output.append(stack.pop()) return ''.join(output)

Binary Tree Traversal

Inorder, Preorder, and Postorder tree traversal methods

Tree Traversal

class Node: def __init__(self, value): self.value = value self.left = None self.right = None def inorder(node): if node: inorder(node.left) print(node.value, end=' ') inorder(node.right) def preorder(node): if node: print(node.value, end=' ') preorder(node.left) preorder(node.right) def postorder(node): if node: postorder(node.left) postorder(node.right) print(node.value, end=' ')
